<script>
let reactionScore = 0;
let memoryScore = 0;

// Reaction Test
function startReaction() {
    reactionScore = 0;
    spawn();
}

function spawn() {
    let target = document.getElementById("target");
    let x = Math.random() * (window.innerWidth - 100);
    let y = Math.random() * (window.innerHeight - 100);

    target.style.left = x + "px";
    target.style.top = y + "px";
    target.style.display = "block";

    target.onclick = function() {
        reactionScore++;
        document.getElementById("reactionScore").innerText =
            "Hits: " + reactionScore;
        target.style.display = "none";
        setTimeout(spawn, 800);
    };

    setTimeout(() => {
        target.style.display = "none";
        updateFinal();
    }, 8000);
}

// Memory Test
function startMemory() {
    let grid = document.getElementById("memoryGrid");
    grid.innerHTML = "";
    memoryScore = 0;

    let pattern = Math.floor(Math.random() * 6);

    for (let i = 0; i < 6; i++) {
        let box = document.createElement("div");
        box.className = "memory-box";

        if (i === pattern) {
            setTimeout(() => box.style.background = "green", 300);
            setTimeout(() => box.style.background = "#ccc", 1000);
        }

        box.onclick = function() {
            if (i === pattern) {
                memoryScore++;
                document.getElementById("memoryScore").innerText =
                    "Correct!";
            } else {
                document.getElementById("memoryScore").innerText =
                    "Wrong!";
            }
            updateFinal();
        };

        grid.appendChild(box);
    }
}

// Final Result
function updateFinal() {
    let result = "";

    if (reactionScore > 5 && memoryScore > 0) {
        result = "Strong attention performance.";
    } else if (reactionScore > 2) {
        result = "Moderate attention patterns.";
    } else {
        result = "Low attention indicators detected.";
    }

    document.getElementById("finalResult").innerText = result;
}
</script>
